Напишіть програму, яка запитує у користувача номер місяця та виводить на екран кількість днів у цьому місяці. Якщо користувач вводить неправильне значення (номер місяця не з діапазону 1-12), програма повинна обробити виключення та вивести на екран повідомлення про помилку.

Код може виглядати так:

java
Copy code
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        try {
            System.out.print("Введіть номер місяця: ");
            int monthNumber = scanner.nextInt();

            int daysInMonth = getDaysInMonth(monthNumber);
            System.out.println("Кількість днів у місяці: " + daysInMonth);
        } catch (Exception e) {
            System.out.println("Помилка: " + e.getMessage());
        }
    }

    public static int getDaysInMonth(int monthNumber) {
        if (monthNumber < 1 || monthNumber > 12) {
            throw new IllegalArgumentException("Номер місяця повинен бути в діапазоні 1-12");
        }

        int[] daysInMonth = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        return daysInMonth[monthNumber - 1];
    }
}

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        try {
            System.out.print("Введіть перше число: ");
            int num1 = scanner.nextInt();

            System.out.print("Введіть друге число: ");
            int num2 = scanner.nextInt();

            if (num2 == 0) {
                throw new ArithmeticException("Друге число не може дорівнювати нулю");
            }

            int result = num1 / num2;
            System.out.println("Результат ділення: " + result);
        } catch (Exception e) {
            System.out.println("Помилка: " + e.getMessage());
        }
    }
}
